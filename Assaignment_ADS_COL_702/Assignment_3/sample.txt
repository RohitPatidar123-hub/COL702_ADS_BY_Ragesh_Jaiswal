# Variables
CXX = g++           # C++ compiler
CXXFLAGS = -Wall -g  # Compiler flags: show all warnings and debug info
TARGET = checkmatrix # Name of the output executable
SRC = hw3_prog.cpp   # Source file

# Default target: compile and run
all: $(TARGET)
	./$(TARGET)

# Rule to create the executable
$(TARGET): $(SRC)
	$(CXX) $(CXXFLAGS) -o $(TARGET) $(SRC)

# Clean up object and executable files
clean:
	rm -f $(TARGET)
